Відповідь на питання:
1. Які є види циклів?
    for, while, do-while

2. Різниця між while та do while?
    while: Умова перевіряється перед виконанням тіла циклу. 
    Якщо умова спочатку хибна, тіло циклу не виконається жодного разу.
    
    do-while: Умова перевіряється після виконання тіла циклу. 
    Це означає, що тіло циклу виконається принаймні один раз, навіть 
    якщо умова спочатку хибна.    


3. Як записується цикл for?

    for (int i = 1; i <= 10; i++) {
    cout << i << " ";
    }


Завдання 1

#include <iostream>
#include <cmath>  // Для функцій sin і abs

using namespace std;

int main() {
    // Оголошуємо змінні
    double x, y;

    // Введення та обчислення для п'яти значень x
    for (int i = 1; i <= 5; i++) {
        // Введення значення x
        cout << "Введіть значення x для випадку " << i << ": ";
        cin >> x;

        // Обчислення значення функції y
        y = pow(sin(x), 5) + fabs(5 * x - 1.5);

        // Виведення результату
        cout << "Значення y для x = " << x << " дорівнює: " << y << endl;
    }

    return 0;
}

Завдання 2

#include <iostream>

using namespace std;

int main() {
    long long product = 1;  // Змінна для збереження добутку

    // Цикл з параметром для перебору чисел від 1 до 20
    for (int i = 2; i <= 20; i += 2) {  // Парні числа починаються з 2 і зростають на 2
        product *= i;  // Множимо кожне парне число на результат
    }

    // Виведення результату
    cout << "Добуток парних чисел від 1 до 20: " << product << endl;

    return 0;
}

#include <iostream>
#include <iomanip>  // Для форматування виведення
#include <cmath>    // Для математичних функцій

using namespace std;

int main() {
    double a, b, h, x, y;

    // Введення значень a, b і h з клавіатури
    cout << "Введіть початок відрізка a: ";
    cin >> a;
    cout << "Введіть кінець відрізка b: ";
    cin >> b;
    cout << "Введіть крок h: ";
    cin >> h;

    // Встановлюємо початкове значення x
    x = a;

    // Виведення заголовка таблиці
    cout << "---------------------------" << endl;
    cout << ":    X      :     Y        :" << endl;
    cout << "---------------------------" << endl;

    // Цикл з передумовою для табулювання функції
    while (x <= b) {
        // Перевіряємо ОДЗ: x не повинен дорівнювати -7
        if (x == -7) {
            cout << "Значення x = -7 не належить ОДЗ. Припиняємо обчислення." << endl;
            break;
        }

        // Обчислення значення функції y
        y = (5 * x - 7 * x + 3) / (x + 7) - 13 * cos(x);

        // Виведення результатів у вигляді таблиці
        cout << fixed << setprecision(5); // Форматування для 5 знаків після коми
        cout << ": " << setw(8) << x << " : " << setw(10) << y << " :" << endl;

        // Збільшуємо x на крок h
        x += h;
    }

    cout << "---------------------------" << endl;

    return 0;
}
