#include <iostream>

using namespace std;

int main() {
    // Задаємо змінні x, b, c
    double x, b, c, Y;
    
    // Введення значень x, b, c
    cout << "Введіть значення x: ";
    cin >> x;
    cout << "Введіть значення b: ";
    cin >> b;
    cout << "Введіть значення c: ";
    cin >> c;

    // Умова для вибору формули
    if (x < 2) {
        // Обчислення Y для випадку x < 2
        Y = (x - 9 * b * x) / (x - 2 * b * x * x);
    } else {
        // Обчислення Y для випадку x >= 2
        Y = (x - 2 * c * x) / (x - 5 * c * x * x * x);
    }

    // Виведення результату
    cout << "Значення Y = " << Y << endl;

    return 0;
}
