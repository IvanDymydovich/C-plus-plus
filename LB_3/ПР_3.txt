1. Що таке умовний оператор?
    
    Умовний оператор — це інструмент, який дозволяє виконувати різні частини коду залежно
    від виконання певної умови. Умовний оператор перевіряє вираз або умову і, залежно 
    від того, чи вона істинна або хибна, виконує відповідну частину коду.

    if, if-else, switch

#include <iostream>

using namespace std;

int main() {
    int N;

    // Введення тризначного числа N
    cout << "Введіть тризначне число: ";
    cin >> N;

    // Перевірка, чи число тризначне
    if (N < 100 || N > 999) {
        cout << "Помилка: введене число не є тризначним." << endl;
        return 1; // Завершуємо програму, якщо число не тризначне
    }

    // Виділяємо сотні, десятки та одиниці
    int hundreds = N / 100;         // Сотні
    int tens = (N / 10) % 10;       // Десятки
    int ones = N % 10;              // Одиниці

    // Знаходимо кількість парних цифр
    int evenCount = 0;

    if (hundreds % 2 == 0) evenCount++;
    if (tens % 2 == 0) evenCount++;
    if (ones % 2 == 0) evenCount++;

    // Виведення результату
    cout << "Кількість парних цифр: " << evenCount << endl;

    return 0;
}

#include <iostream>

using namespace std;

int main() {
    int tramNumber;

    // Введення номера трамвая
    cout << "Введіть номер трамвая (1-7): ";
    cin >> tramNumber;

    // Використовуємо оператор switch для вибору маршруту
    switch (tramNumber) {
        case 1:
            cout << "Трамвай №1: Початкова зупинка — Площа Льва Толстого, кінцева зупинка — Площа Перемоги." << endl;
            break;
        case 2:
            cout << "Трамвай №2: Початкова зупинка — Вокзальна, кінцева зупинка — Дарниця." << endl;
            break;
        case 3:
            cout << "Трамвай №3: Початкова зупинка — Контрактова площа, кінцева зупинка — Лісова." << endl;
            break;
        case 4:
            cout << "Трамвай №4: Початкова зупинка — Оболонь, кінцева зупинка — Шевченківська." << endl;
            break;
        case 5:
            cout << "Трамвай №5: Початкова зупинка — Либідська, кінцева зупинка — Берестейська." << endl;
            break;
        case 6:
            cout << "Трамвай №6: Початкова зупинка — Троєщина, кінцева зупинка — Відрадний." << endl;
            break;
        case 7:
            cout << "Трамвай №7: Початкова зупинка — Печерськ, кінцева зупинка — Позняки." << endl;

}
